1)Create the Docker file (inside the model folder)
touch Dockerfile
vim Dockerfile #to edit go in to insert mode
FROM continuumio/anaconda3:4.4.0 #python conda dependencies
COPY . /usr/app/  #current working directory
EXPOSE 5000 #port no
WORKDIR /usr/app/  #set the working directory(same as previous)
RUN pip install -r requirements.txt  #install model requirements in to the image
CMD python flask_api.py   #run api file


2)Build the docker image
eg-
docker build -t xxxx .

3)Run that docker image
docker run -p 8000:8000 xxxx
